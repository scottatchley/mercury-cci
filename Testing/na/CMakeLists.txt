function(build_na_test test_name)
  add_executable(${test_name} ${test_name}.c)
  target_link_libraries(${test_name} mercury_test)
  if(MERCURY_ENABLE_COVERAGE)
    set_coverage_flags(${test_name})
  endif()
endfunction()
